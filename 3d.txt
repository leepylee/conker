func main()
  num oy = -1.5 // 1.5 M below camera
end

func num displacement(num zDiff, num xOrY)
  num ratio = xOrY / zDiff
  num disp = ratio * 1000 
  return disp
end

func draw()
  paint BLUE
  strokewidth 3
  plot MIDX, MIDY
  paint WHITE
  for zd from 1 to 100 step 10 // from 1M to 100M away from camera
    for xd from -5 to 5 // from left to right
      num dispY = displacement(zd, oy)
      num dispX = displacement(zd, xd)
      plot MIDX - dispX, MIDY - dispY
    end
  end 
end
